document.addEventListener('DOMContentLoaded', function() { // Remplacer 'data.json' par le chemin exact si votre fichier est dans un sous-dossier fetch('data.json') .then(response => { if (!response.ok) { throw new Error('Network response was not ok'); } return response.json(); }) .then(rawData => { // Transformation des données brutes en un format structuré pour l'application const data = rawData.reduce((acc, item) => { if (!acc[item.Theme]) acc[item.Theme] = {}; if (!acc[item.Theme][item.Chapitre]) acc[item.Theme][item.Chapitre] = []; acc[item.Theme][item.Chapitre].push({ question: item.Question, reponse: item.Reponse }); return acc; }, {}); function initThemes() { const themeSelect = document.getElementById('theme'); themeSelect.innerHTML = ''; Object.keys(data).forEach(theme => { const option = document.createElement('option'); option.value = theme; option.text = theme; themeSelect.add(option); }); populateChapters(); } function populateChapters() { const theme = document.getElementById('theme').value; const chapters = document.getElementById('chapter'); chapters.innerHTML = ''; Object.keys(data[theme]).forEach(chapter => { const option = document.createElement('option'); option.value = chapter; option.text = chapter; chapters.add(option); }); } function getRandomQuestionAnswer(theme, chapter) { const questions = data[theme][chapter]; const randomIndex = Math.floor(Math.random() * questions.length); return questions[randomIndex]; } document.getElementById('theme').addEventListener('change', populateChapters); document.getElementById('questionBtn').addEventListener('click', () => { const theme = document.getElementById('theme').value; const chapter = document.getElementById('chapter').value; const qa = getRandomQuestionAnswer(theme, chapter); document.getElementById('questionArea').innerText = `Question: ${qa.question}`; document.getElementById('questionArea').style.display = 'block'; }); document.getElementById('answerBtn').addEventListener('click', () => { if (document.getElementById('questionArea').style.display === 'block') { const theme = document.getElementById('theme').value; const chapter = document.getElementById('chapter').value; const qa = getRandomQuestionAnswer(theme, chapter); document.getElementById('answerArea').innerText = `Answer: ${qa.reponse}`; document.getElementById('answerArea').style.display = 'block'; } }); initThemes(); // Initialise les thèmes et les chapitres au chargement de la page }) .catch(error => console.error('Error loading the JSON file:', error)); });
